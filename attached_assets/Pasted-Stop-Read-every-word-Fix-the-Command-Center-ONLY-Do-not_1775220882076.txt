FIX 1 — ADD STATS ENDPOINT
═══════════════════════════════════════════════════════════

In dashboard_briefing.ts add this route:

router.get("/dashboard/stats", authMiddleware, async (req, res) => {
  const userId = req.user.id;

  const memberships = await db
    .select({ projectId: projectMembers.projectId })
    .from(projectMembers)
    .where(eq(projectMembers.userId, userId));

  const projectIds = memberships.map(m => m.projectId);

  if (projectIds.length === 0) {
    return res.json({
      activeProjects: 0,
      filesProcessed: 0,
      openRfis: 0,
      pendingSubmittals: 0,
      complianceRate: null,
      filesNeedingAttention: 0
    });
  }

  const [
    activeProjectsRes,
    allFilesRes,
    compliantFilesRes,
    attentionFilesRes,
    openRfisRes,
    pendingSubmittalsRes
  ] = await Promise.all([
    db.select({ id: projects.id }).from(projects)
      .where(and(
        inArray(projects.id, projectIds),
        ne(projects.status, 'archived')
      )),
    db.select({ id: files.id }).from(files)
      .where(inArray(files.projectId, projectIds)),
    db.select({ id: files.id }).from(files)
      .where(and(
        inArray(files.projectId, projectIds),
        eq(files.isCompliant, true)
      )),
    db.select({ id: files.id }).from(files)
      .where(and(
        inArray(files.projectId, projectIds),
        eq(files.isCompliant, false)
      )),
    db.select({ id: rfis.id }).from(rfis)
      .where(and(
        inArray(rfis.projectId, projectIds),
        ne(rfis.status, 'closed')
      )),
    db.select({ id: submittals.id }).from(submittals)
      .where(and(
        inArray(submittals.projectId, projectIds),
        inArray(submittals.status, ['pending', 'under_review'])
      ))
  ]);

  const totalFiles = allFilesRes.length;
  const complianceRate = totalFiles > 0
    ? Math.round((compliantFilesRes.length / totalFiles) * 100)
    : null;

  res.json({
    activeProjects: activeProjectsRes.length,
    filesProcessed: totalFiles,
    openRfis: openRfisRes.length,
    pendingSubmittals: pendingSubmittalsRes.length,
    complianceRate,
    filesNeedingAttention: attentionFilesRes.length
  });
});

async function verifyReportToken(req: any, res: any): 
Promise<number | null> {
  const token = 
    req.headers.authorization?.split(' ')[1] || 
    (req.query.token as string);
  if (!token) {
    res.status(401).json({ error: 'Authentication required' });
    return null;
  }
  try {
    const decoded = jwt.verify(
      token, process.env.JWT_SECRET!) as any;
    return decoded.userId || decoded.id;
  } catch {
    res.status(401).json({ error: 'Invalid token' });
    return null;
  }
}
