router.post("/projects/:projectId/rfis/import",
  authMiddleware,
  requirePermission("admin", "write"),
  multer({ storage: multer.memoryStorage(), limits: { fileSize: 50 * 1024 * 1024 } }).single("file"),
  async (req, res) => {
    const projectId = Number(req.params.projectId);
    try {
      if (!req.file) { res.status(400).json({ error: "no_file" }); return; }
      const anthropic = new Anthropic({
        apiKey: process.env.AI_INTEGRATIONS_ANTHROPIC_API_KEY ?? "",
        baseURL: process.env.AI_INTEGRATIONS_ANTHROPIC_BASE_URL ?? undefined,
      });
      const fileContent = req.file.buffer.toString("utf-8").slice(0, 15000);
      const msg = await anthropic.messages.create({
        model: "claude-sonnet-4-5",
        max_tokens: 4000,
        messages: [{
          role: "user",
          content: `You are analyzing a construction RFI (Request for Information) log document.
Extract all RFI records from this document.
Document content:
${fileContent}

Return a JSON array of RFI objects. Each object should have these fields (use null if not found):
{
  "number": "RFI number like RFI-001",
  "subject": "RFI subject/title",
  "description": "full description",
  "status": "open/closed/pending",
  "assignedToName": "person responsible",
  "submittedByCompany": "company that submitted",
  "dueDate": "due date string or null",
  "discipline": "trade/discipline"
}
Return ONLY a JSON array. No markdown. No explanation.`
        }]
      });
      const text = msg.content[0]?.type === "text" ? msg.content[0].text : "[]";
      const clean = text.replace(/```json\n?|```/g, "").trim();
      const records = JSON.parse(clean) as any[];
      let imported = 0;
      for (const r of records) {
        if (!r.subject && !r.number) continue;
        await db.insert(rfisTable).values({
          projectId,
          number: r.number || `RFI-${String(imported + 1).padStart(3, "0")}`,
          subject: r.subject || "Imported RFI",
          description: r.description || null,
          status: r.status || "open",
          assignedToId: null,
          submittedById: req.user!.userId,
          submittedByCompany: r.submittedByCompany || null,
          dueDate: r.dueDate ? new Date(r.dueDate) : null,
          discipline: r.discipline || null,
        });
        imported++;
      }
      await db.insert(activityLogTable).values({
        projectId,
        userId: req.user!.userId,
        userFullName: req.user!.fullName ?? "",
        userCompanyName: req.user!.companyName ?? "",
        actionType: "import",
        entityType: "rfi",
        entityId: projectId,
        details: `Imported ${imported} RFIs from ${req.file.originalname}`,
      });
      res.json({ imported, message: `${imported} RFIs imported successfully` });
    } catch (err) {
      console.error("[rfi-import] FAILED:", err);
      res.status(500).json({ error: err instanceof Error ? err.message : String(err) });
    }
  }
);
